\slide{qf}{
    Ecrires les nombres suivant sous forme décimale :
    \multiColEnumerate{2}{
        \item $\frac{5}{2} = \awsr{\np{2.5}}$
        \item $\frac{9 \times 6}{3} = \awsr{27}$
        \item $\frac{4 + 6}{6-11} = \awsr{-2}$
        \item $\frac{6 \times (-6)}{-100 \div 10} = \awsr{\np{3.6}}$
    }
}

\slide{cr}{
    \df{}{
        Un nombre rationnel est, en mathématiques,
        un nombre qui peut s'exprimer comme le quotient de deux entiers relatifs.
    }
}

\slide{cr}{
    Pour $a,b,c,d$ et $n$ des nombres.

    \pr{Somme ou différence de fractions avec le \key{même dénominateur}}{
        \begin{enumerate}
            \item On prend la somme ou différence des numérateurs.
            \item On conserve le même dénominateur.
        \end{enumerate}
    }

    \expl{}{
        \multiColEnumerate{2}{
            \item $\frac{1}{2} + \frac{5}{2}
            = \awsr{\frac{1+5}{2} = \frac{6}{2}}$
            \item $\frac{14}{21} - \frac{12}{21}
            = \awsr{\frac{14-12}{21} = \frac{2}{21}}$
            \item $\frac{a}{d} + \frac{b}{d}
            = \awsr{\frac{a+b}{d}}$
            \item $\frac{a}{d} - \frac{b}{d}
            = \awsr{\frac{a-b}{d}}$
        }
    }

    \mthd{Somme et différence de fractions de \key{dénominateur différent}}{
        \begin{enumerate}
            \item On met les fractions au même dénominateur.
            \item On fait leur somme ou différence comme précedement.
        \end{enumerate}
    }

    \expl{}{
        \multiColEnumerate{1}{
            \item $\frac{1}{2} - \frac{1}{4}
            = \awsr{\frac{2}{4} - \frac{1}{4}
            = \frac{2-1}{4} = \frac{1}{4}
            }$
            \item $\frac{2}{12} + \frac{2}{3}
            = \awsr{\frac{2}{12} + \frac{2\times4}{3\times4}
            = \frac{2}{12} + \frac{8}{12}
            = \frac{2+8}{12}
            = \frac{10}{12}
            }$
            \item $\frac{3}{5} + \frac{6}{7}
            = \awsr{\frac{3\times7}{5\times7} + \frac{6\times5}{7\times5}
            = \frac{21}{35} + \frac{30}{35}
            = \frac{21+30}{35} = \frac{51}{35}
            }$
            \item $\frac{12}{3} - \frac{6}{2}
            = \awsr{\frac{12\times2}{3\times2} - \frac{6\times3}{2\times3}
            = \frac{24}{6} - \frac{18}{6}
            = \frac{24-18}{6} = \frac{6}{6}
            }$
            \item $\frac{a}{b} + \frac{c}{d}
            = \awsr{\frac{a\times d}{b\times d} + \frac{c \times b}{d\times b}
            = \frac{ad}{bd} + \frac{cb}{bd}
            = \frac{ad+cb}{bd}
            }$
        }
    }
}

\slide{cr}{
    \pr{Produits de fractions}{
        $\frac{a}{c} \times \frac{b}{d} = \frac{a \times b}{c \times d}$
    }
}
\slide{cr}{
    \rmk{} {
        On appelle \key{inverse} d'un nombre $x$, le nombre qui, multiplié par $x$, donne $1$.
    }[\wiki{Inverse}]
    
    \expl{} {
        \multiColEnumerate{1}{
            \item $0.5 \times \awsr{2} = 1$. L'inverse de $0.5$ est donc $\awsr{2}$.
            \item $10 \times \awsr{0.1} = 1$. L'inverse de $10$ est donc $\awsr{0.1}$.
            \item $\frac{1}{16} \times \awsr{16} = 1$. L'inverse de $\frac{1}{9}$ est donc $\awsr{9}$.
            \item $23 \times \awsr{\frac{1}{23}} = 1$. L'inverse de $23$ est donc $\awsr{\frac{1}{23}}$.
            \item $\frac{4}{5} \times \awsr{\frac{5}{4}} = 1$. L'inverse de $\frac{4}{5}$ est donc $\awsr{\frac{5}{4}}$.
        }
    }
    
    \rmk{}{
        \begin{enumerate}
            \item L'inverse d'un nombre $x$ est $\awsr{\frac{1}{x}}$.
            \item L'inverse d'un rationnel $\frac{n}{d}$, le rationnel $\awsr{\frac{d}{n}}$.
        \end{enumerate}
    }

    \pr{Division par un rationnels}{
        Diviser un nombre par un rationnel est équivalent à le multiplier par l'inverse de ce rationnel.
        $a \div \frac{n}{d} = a \times \frac{d}{n}$ 
    }
}
