\slide{cr}{\bsmall
    \sseq\ssec
    \df{}{
        Deux angles formés par deux droites coupées par une sécante sont dits \key{alternes-internes} si :
        \begin{itemize}
            \item ils sont situés de part et d'autre de la sécante (\key{alternes});
            \item ils sont situés entre les deux droites (\key{internes});
            \item ils ne sont pas sur le même sommet.
        \end{itemize}
    }[\wiki{Angles_alternes-internes}]
}

        \begin{itemize}
            \item ils sont du même côté de la sécante;
            \item l'un est situé entre les deux droites et l'autre hors des deux droites;

\scn{Somme des angles dans un triangle}
\avspace{1cm}
\bookSlide{33p298,38p298}[7cm][2]

\nswr[0]{\textbf{33p298} :\setItemColor{myriade}
    \begin{enumerate}
        \item \begin{itemize}
            \item On sait que $(AC)$ et $(ED)$ sont parallèles, 
            et que $\widehat{ABE}$ et $\widehat{BED}$ sont deux angles alternes-internes.
            \item Or, deux angles alternes-internes formés par deux droites parallèles sont égaux.
            \item Donc, $\widehat{ABE} = \widehat{BED}$.
        \end{itemize}
        %
        \item Par un raisonnement similaire appliqué aux angles $\widehat{CBD}$ et $\widehat{BDE}$, 
        on obtient $\widehat{CBD} = \widehat{BDE}$.
        %
        \item \begin{itemize}
            \item On a $\widehat{ABC} = \widehat{ABE} + \widehat{DBE} + \widehat{CBD}$.
            \item En utilisant les résultats des questions précédentes, on en déduit que : \\
            $\widehat{ABC} = \widehat{BED} + \widehat{DBE} + \widehat{BDE} = \widehat{BED} + \widehat{EDB} + \widehat{DBE}$.
            \item Or, $\widehat{ABC}$ est un angle plat, et sa mesure est donc $\ang{180}$.
            \item Ainsi, $\widehat{BED} + \widehat{EDB} + \widehat{DBE} = \ang{180}$.
        \end{itemize}
    \end{enumerate}
}
