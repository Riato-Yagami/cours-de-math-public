\bsubsec{Division euclidienne}

\slide{exo}{\bshrink
    \act{}{%
    Le roi de Divisia possède $27$ pièces d'or et souhaite les partager équitablement entre $4$ chevaliers.
    Les pièces restantes seront données à son écuyer.
    
    \begin{enumerate}
        \item Combien de pièces chaque chevalier recevra-t-il ?
        \item Combien de pièces resteront pour l'écuyer ?
        \item Supposons maintenant que le roi possède $100$ pièces et qu'il les partage entre $6$ chevaliers :
        \begin{enumerate}
            \item Combien de pièces chaque chevalier recevra-t-il ?
            \item Combien de pièces resteront pour l'écuyer ?
        \end{enumerate}\saveenumi
    \end{enumerate}
    }
}

\slide{exo}{
    \begin{enumerate}\loadenumi[act]
        \item Toujours avec $100$ pièces, existe-t-il un nombre de chevaliers tel que :
        \begin{enumerate}
            \item l'écuyer ne reçoive aucune pièce ?
            \item l'écuyer reçoive plus de pièces qu'un chevalier ?
            \item l'écuyer reçoive exactement autant de pièces qu'un chevalier ?
        \end{enumerate}
    \end{enumerate}
}

\slide{cr}{\bsmall
    \ssubsec

    \df{}{
        Pour deux entier $a$ et $b$,
        on appelle \key{division euclidienne}
        du \textcolor{Green}{\key{dividende} $a$} par le \textcolor{Red}{\key{diviseur} $b$}
        l'expression :
        \begin{align*}
            \textcolor{Green}{a}
            = \textcolor{Red}{b}
            \times \textcolor{Blue}{q}
            + \textcolor{Violet}{r}
        \end{align*}
        Où le \textcolor{Blue}{\key{quotients} $q$} et \textcolor{Violet}{\key{reste} $r$} sont deux entiers avec $r < b$.
    }

    \expl{}{
        \multiColEnumerate{3}{
            \item $100 = 6 \times \bawsr{16} + \bawsr{4}$
            \item $246 = 3 \times \bawsr{82} + \bawsr{0}$
            \item $360 = 23 \times \bawsr{15} + \bawsr{15}$
        }
    }
}

\slide{cr}{\bsmall
    \df{}{
        Pour $a$ et $b$ deux entiers,
        on dit que $b$ \key{divise} $a$ si le reste de la division euclidienne de $a$ par $b$ est $0$.
    }

    \pr{}{
        \Sialors{$b$ \key{divise} $a$}{alors $a$ est un \key{multiple} de $b$}.
    }

    \begin{center}
        \expl{}{
