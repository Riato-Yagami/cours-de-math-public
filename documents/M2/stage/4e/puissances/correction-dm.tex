    \begin{enumerate}[label=\textbf{\color{DarkOrchid}\arabic*.}]\setlength{\itemsep}{15pt}%
        \item
        \begin{itemize}%
            \item Pour deux bonnes réponses on multiplie le premier gain de $1\euro$ par $2$.\\
            Le gain est donc de $1 \times 2 = 2 \euro$.
            %
            \item Pour cinq bonnes réponses on multiplie le premier gain par $2$, $4$ fois.\\
            Le gain est donc de $1 \times \pow{2}{4} = \pow{2}{4} = 2^4 = 16\euro$.
        \end{itemize}%
        %
        \item Pour 20 bonnes réponses on multiplie le premier gain par $2$, $19$ fois.\\
        Le gain est donc de $1 \times 2^{19} = 2^{19} = \num{524 288}\euro$.
        %
        \item On a $2^{14} = \num{16 384} \et 2^{15} = \num{32 768}$\\
        Le gain pour avoir donné $14$ bonnes réponses est donc de $\num{16 384}\euro$,
        et celui pour avoir donné $15$ bonnes réponses de $\num{32 768}\euro$.\\
        Or $\num{16 384} < \num{20 000} < \num{32 768}$\\
        Il n'y a donc pas de nombre de bonne réponse associé au gain de $\num{20 000}\euro$.
    \end{enumerate}
