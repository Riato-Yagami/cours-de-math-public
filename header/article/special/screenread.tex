\renewcommand{\slide}[2]{%
    \setPhase{#1}%
    \xdef\currentPhase{\phase}%
    \ifthenelse{\boolean{showAllPhases}}{%
        \begin{center}%
            \begin{phaseBox}%
                \textbf{\phase \hfill \longTitle}
            \end{phaseBox}%
        \end{center}}{%
    }%
    \ifthenelse{\boolean{showAllPhases}}{#2}{%
        \ifthenelse{\equal{\phase}{\showPhase}}{#2}{}%
    }%
    \foot%
    \pbreak%
}
