        \item Ranger dans l'odre croissant les nombres décimaux suivants :
        \multiColItemize{2}{
            \item $\frac{6}{10} + \frac{1}{100} + \frac{1}{10000}$

            \item $\np{6.1111}$
            \item $6+ \frac{101}{1000}$
            \item $6111$ dix-millièmes
            \item $\frac{6101}{10000}$
        }
        \item Combien de nombres différents sont écrits dans la liste ci-dessous :
        \multiColItemize{5}{
            \item $\frac{1284}{10000}$
            \item $\frac{1}{4}$
            \item $\np{1.4}$
            \item $\np{0.25}$
            \item $\frac{25}{100}$
        }
    \end{enumerate}
