\act{Multiplication de décimaux}{
    \begin{enumerate}
        \item \begin{enumerate}
            \item $1 \textrm{ dixième} \times 1 \textrm{ dixième} = \nswr{1} \textrm{ centième}$
            \item $15 \textrm{ dixièmes} \times 2 \textrm{ dixièmes} = 30 \nswr{\textrm{ centièmes}}$
            \item $3 \textrm{ dixièmes} \times 4 \textrm{ centièmes} = \nswr{12 \textrm{ millièmes}}$
            \item $91 \textrm{ centièmes} \times 5 \textrm{ millièmes} = \nswr{455 \textrm{ centièmes de millième}}$
        \end{enumerate}
        \item \multiColEnumerate{2}{
            \item $\np{0.1} \times \np{0.1} = \nswr{\np{0.01}}$
            \item $\np{1.5} \times \np{0.2} = \nswr{\np{0.3}}$
            \item $\np{0.3} \times \np{0.04} = \nswr{\np{0.012}}$
            \item $\np{0.91} \times \np{0.005} = \nswr{\np{0.00450}}$
        }
        \item \multiColEnumerate{1}{
            \item $\np{0.430} \times \np{0.2} = 430 \nswr{\textrm{ millièmes}} \times 2 \nswr{\textrm{ dixièmes}}$\\
            $= 860 \nswr{\textrm{ dix-millièmes}} = \nswr{\np{0.0860}}$
            \item $\nswr{\np{0.00103}} \times \nswr{\np{1.1}} = 103 \textrm{ cent-millièmes} \times 11 \textrm{ dixièmes}$\\
            $= 1133 \nswr{\textrm{ millionièmes}} = \nswr{\np{0.001133}}$
            \item $\np{21.3} \times \np{0.04} = \nswr{213 \textrm{ dixièmes}} \times \nswr{4 \textrm{ centièmes}}$\\
            $= \nswr{852 \textrm{ millièmes}} = \nswr{\np{0.852}}$
        }
    \end{enumerate}
}
