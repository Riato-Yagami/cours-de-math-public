\exo{Développer}{
    Développer les expressions suivantes :
    \begin{enumerate}
        \item $\np{3.1} \times (9+4) =
        \nswr{\np{3.1} \times 9 + \np{3.1} \times 4}$
        \item $x \times (6+3) =
        \nswr{x \times 6 + x \times 3}$
        \item $602 \times (k+7) =
        \nswr{602 \times k + 602 \times 7}$
    \end{enumerate}
}
