\exo{Simplifier des sommes}{
    Simplifier les expressions suivantes :
        \multiColEnumerate{2}{
            \item $3x + 5x = \nswr{8x}$
            \item $7y - 9y + y = \nswr{-y}$
            \item $4a + 3b - a + 2b = \nswr{3a + 5b}$
            \item $6m^2 - 3m^2 + 2m = \nswr{3m^2 + 2m}$
            \item $2a + 4 + a =\nswr{3a + 4}$
            \item $5pq + 3q - 2p + q = \nswr{5pq + 4q - 2p}$
        }
}
