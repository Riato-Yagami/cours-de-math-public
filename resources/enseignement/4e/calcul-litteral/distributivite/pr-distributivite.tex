\def\dk{\textcolor{Green}{k}}

\pr{Distributivité}{
    La \mlt[multiplication] est \key{distributive} par rapport à l'\add[addition] ;
    pour tous nombres $a$, $b$ et $k$, on a :
    \begin{align*}
        \dk \mlt (a \add b) &= (\dk \mlt a) \add (\dk \mlt b)
    \end{align*}
