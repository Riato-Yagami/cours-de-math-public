\exo{Distributivité simple}{
    Développer et réduire les expressions suivantes :
    \begin{enumerate}
        \item $\np{3} \times (9z+4) =
        \nswr[2]{\np{3} \times 9z + \np{3} \times 4 = 27z+12}$
        \item $x \times (6+3x) =
        \nswr[2]{x \times 6 + x \times 3x = 6x+3x^2}$
        \item $60y \times (k+7) =
        \nswr[2]{60y \times k + 60y \times 7 = 60yk+420y}$
    \end{enumerate}
}
