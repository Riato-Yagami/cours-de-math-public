\exo{Factoriser}{
    Factorise les expressions suivantes :
    \begin{enumerate}
        \item $16 \times \np{4.9} - 6 \times \np{4.9}
        = \nswr[3]{\np{4.9} \times (16 - 6)}$
        \item $3x - 9
        = \nswr[3]{3(x - 3)}$
        \item $15 + 45y
        = \nswr[3]{(1 + 3y)}$
        % \item $31z - 31
        % = \nswr[3]{31(z - 1)}$
    \end{enumerate}
