\df{}{
    On appelle \key{moyenne} d'une série statistique le nombre obtenu en additionnant toutes les valeurs de cette série,
    puis en divisant par l'effectif total.
    \begin{align*}
        \textrm{Moyenne} = \frac{\textrm{Somme des valeurs}}{\textrm{Effectif total}}
    \end{align*}
}[\wiki{Moyenne}]
