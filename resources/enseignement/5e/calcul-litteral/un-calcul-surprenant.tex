\exo{Un calcul surprenant}{
    \noCalculator
    \begin{enumerate}
        \item Calculer :\multiColEnumerate{3}{
            \item $1 \times 3 - 2^2$.
            \item $2 \times 4 - 3^2$.
            \item $3 \times 5 - 4^2$.
        }
        \item Conjecturer le resultat de $317 \times 319 - 318^2$.
        \item Proposer une expression sur ce même modèle mettant en jeu trois autres nombres entiers.
        \item Le professeur généralise ces calculs en proposant l'expression $(n - 1)(n + 1) - n^2$.
        Que représente $n$ ?
        QCM :
        \multiColItemize{1}{
            \item le plus petit des nombres?
            \item le nombre du milieu?
            \item le plus grand des nombres?
        }
    \end{enumerate}
