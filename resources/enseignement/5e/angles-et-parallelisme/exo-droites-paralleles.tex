\exo{Droites parallèles}{
    Sur la figure ci-dessous, les droites $(AB)$ et $(DC)$ sont parallèles.
    \input{resources/enseignement/5e/angles-et-parallelisme/tikz/droite-paralleles-angles-alternes-internes.tex}
    Quelle est la mesure de l'angle $\widehat{DHG}$ ?
}

\nswr[0]{
    \begin{itemize}
        \item On a les droites $(AB)$ et $(DC)$ parallèles.
        \item Et les angles $\widehat{HGB}$ et $\widehat{DHG}$ sont alternes-internes.
        \item Or deux angles alternes-internes freposant sur deux droites parallèles et une sécante sont égaux.
        \item Alors $\widehat{DHG} = \widehat{HGB}$.
        \item Donc, la mesure de l'angle $\widehat{DHG}$ est également de $\ang{112}$ degrés.
    \end{itemize}
}
